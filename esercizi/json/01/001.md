```sql
-- ===============================================
-- CREAZIONE TABELLE CON CAMPI JSON
-- ===============================================

-- Tabella per profili utente con dati JSON
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(100),
    profile JSON,
    preferences JSONB,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Tabella per prodotti e-commerce
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200),
    specifications JSONB,
    reviews JSONB[],  -- Array di oggetti JSON
    metadata JSON
);

-- Tabella per eventi con dati strutturati
CREATE TABLE events (
    id SERIAL PRIMARY KEY,
    event_name VARCHAR(100),
    event_data JSONB,
    participants JSONB
);

```
